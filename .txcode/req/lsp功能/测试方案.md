### 2.2 数据库测试

**文件**: `test/lsp/sql.test.ts`

```typescript
import { describe, test, expect, beforeEach, afterEach } from "jest";
import Database from "better-sqlite3";
import { DbService } from "../../src/modules/db/db.service";
import { lspServerTable, defaultLSPServers, insertLSPServerSQL } from "../../src/lsp/sql";

describe("Database", () => {
  let dbService: DbService;
  let db: Database.Database;
  
  beforeEach(() => {
    dbService = new DbService(":memory:");
    dbService.init();
    db = dbService.getDb();
  });
  
  afterEach(() => {
    db.close();
  });

  test("initDatabase - 应创建表结构", () => {
    db.run(lspServerTable);
    
    const tables = db.prepare(
      "SELECT name FROM sqlite_master WHERE type='table'"
    ).all();
    
    const tableNames = tables.map((t: any) => t.name);
    expect(tableNames).toContain("lsp_server");
  });

  test("initDatabase - 应创建默认数据", () => {
    db.run(lspServerTable);
    
    const now = Date.now();
    for (const server of defaultLSPServers) {
      db.run(insertLSPServerSQL, [server.id, server.enabled, server.auto_start, now, now]);
    }
    
    const servers = db.prepare("SELECT * FROM lsp_server").all();
    expect(servers.length).toBeGreaterThan(0);
  });
});
```

### 2.3 LSP 管理器测试

**文件**: `test/lsp/manager.test.ts`

```typescript
import { describe, test, expect, beforeEach } from "jest";
import { DbService } from "../../src/modules/db/db.service";
import { LSPManager } from "../../src/lsp/manager";
import { lspServerTable, defaultLSPServers, insertLSPServerSQL } from "../../src/lsp/sql";

describe("LSPManager", () => {
  let dbService: DbService;
  
  beforeEach(async () => {
    dbService = new DbService(":memory:");
    dbService.init();
    
    dbService.run(lspServerTable);
    const now = Date.now();
    for (const server of defaultLSPServers) {
      dbService.run(insertLSPServerSQL, [server.id, server.enabled, server.auto_start, now, now]);
    }
    
    await LSPManager.initDatabase();
  });

  test("loadConfig - 应正确加载配置", async () => {
    const config = await LSPManager.loadConfig();
    
    expect(config.size).toBeGreaterThan(0);
    expect(config.has("typescript")).toBe(true);
    expect(config.get("typescript")?.enabled).toBe(true);
  });

  test("updateServer - 应更新配置", async () => {
    await LSPManager.updateServer("typescript", { enabled: false });
    
    const config = await LSPManager.loadConfig();
    expect(config.get("typescript")?.enabled).toBe(false);
  });

  test("checkServerExists - 应正确检测服务器", async () => {
    const exists = await LSPManager.checkServerExists("nonexistent");
    expect(typeof exists).toBe("boolean");
  });

  test("getServerPath - 应返回正确路径", () => {
    const path = LSPManager.getServerPath("typescript");
    expect(path).toContain("typescript");
  });

  test("checkJavaVersion - 应返回版本检查结果", async () => {
    const result = await LSPManager.checkJavaVersion();
    
    expect(result).toHaveProperty("valid");
    expect(result).toHaveProperty("error");
    expect(typeof result.valid).toBe("boolean");
  });
});
```

### 3.1 服务器启动集成测试

**文件**: `test/lsp/startup.test.ts`

```typescript
import { describe, test, expect, beforeAll, afterAll, beforeEach } from "jest";
import { DbService } from "../../src/modules/db/db.service";
import { LSPManager } from "../../src/lsp/manager";
import { LSPDownloader } from "../../src/lsp/downloader";
import { lspServerTable, defaultLSPServers, insertLSPServerSQL } from "../../src/lsp/sql";

describe("LSP 服务器启动集成测试", () => {
  beforeAll(async () => {
    const dbService = new DbService(":memory:");
    dbService.init();
    
    dbService.run(lspServerTable);
    const now = Date.now();
    for (const server of defaultLSPServers) {
      dbService.run(insertLSPServerSQL, [server.id, server.enabled, server.auto_start, now, now]);
    }
    
    await LSPManager.initDatabase();
  });

  afterAll(async () => {
    await LSPManager.stopAll();
  });

  test("应能获取服务器状态", async () => {
    const status = await LSPManager.getServerStatus("typescript");
    expect(status).toHaveProperty("status");
    expect(status).toHaveProperty("error");
  });

  test("stopServer - 不存在的服务器应正常处理", async () => {
    await expect(LSPManager.stopServer("nonexistent")).resolves.toBeUndefined();
  });

  test("Java 版本检查应返回有效结果", async () => {
    const result = await LSPManager.checkJavaVersion();
    expect(typeof result.valid).toBe("boolean");
  });

  test("未启用服务器启动应返回 false", async () => {
    const started = await LSPManager.startServer("nonexistent");
    expect(started).toBe(false);
  });
});
```

### 3.2 Web API 集成测试

**文件**: `test/api/lsp.test.ts`

```typescript
import { describe, test, expect, beforeAll, afterAll } from "jest";
import express, { Express } from "express";
import { lspRouter } from "../../src/api/lsp";

describe("LSP API", () => {
  let app: Express;
  
  beforeAll(() => {
    app = express();
    app.use(express.json());
    app.use("/api/lsp", lspRouter);
  });

  test("GET /api/lsp/servers - 应返回所有服务器", async () => {
    const response = await require("supertest")(app).get("/api/lsp/servers");
    
    expect(response.status).toBe(200);
    expect(response.body.success).toBe(true);
    expect(Array.isArray(response.body.data)).toBe(true);
    expect(response.body.data.length).toBeGreaterThan(0);
    
    const ts = response.body.data.find((s: any) => s.id === "typescript");
    expect(ts).toHaveProperty("enabled");
    expect(ts).toHaveProperty("status");
  });

  test("PUT /api/lsp/servers/:id - 应更新配置", async () => {
    const response = await require("supertest")(app)
      .put("/api/lsp/servers/typescript")
      .send({ enabled: false });
    
    expect(response.status).toBe(200);
    expect(response.body.success).toBe(true);
  });

  test("POST /api/lsp/servers/:id/start - 无效服务器应返回错误", async () => {
    const response = await require("supertest")(app)
      .post("/api/lsp/servers/nonexistent/start");
    
    expect(response.status).toBe(400);
  });

  test("POST /api/lsp/servers/:id/stop - 无效服务器应正常处理", async () => {
    const response = await require("supertest")(app)
      .post("/api/lsp/servers/nonexistent/stop");
    
    expect(response.status).toBe(200);
  });

  test("GET /api/lsp/servers/:id/status - 应返回状态", async () => {
    const response = await require("supertest")(app)
      .get("/api/lsp/servers/typescript/status");
    
    expect(response.status).toBe(200);
    expect(response.body.success).toBe(true);
    expect(response.body.data).toHaveProperty("status");
  });

  test("GET /api/lsp/java-version - 应返回 Java 版本信息", async () => {
    const response = await require("supertest")(app)
      .get("/api/lsp/java-version");
    
    expect(response.status).toBe(200);
    expect(response.body.success).toBe(true);
    expect(response.body.data).toHaveProperty("valid");
    expect(response.body.data).toHaveProperty("error");
  });
});
```

### 4.1 LspTool 测试

```typescript
import { describe, test, expect } from "jest";
import { lspTool } from "../../src/tool/lsp";

describe("LspTool", () => {
  test("lspTool - 应有正确的名称和描述", () => {
    expect(lspTool.name).toBe("lsp");
    expect(lspTool.description).toContain("LSP");
  });

  test("lspTool - 应有正确的参数定义", () => {
    expect(lspTool.parameters).toHaveProperty("type", "object");
    expect(lspTool.parameters).toHaveProperty("properties");
    expect(lspTool.parameters.properties).toHaveProperty("action");
  });

  test("lspTool.execute - 缺少 action 应返回错误", async () => {
    const result = await lspTool.execute({} as any);
    const parsed = JSON.parse(result);
    expect(parsed).toHaveProperty("error");
  });

  test("lspTool.execute - workspaceSymbol 缺少 query 应返回错误", async () => {
    const result = await lspTool.execute({
      action: "workspaceSymbol"
    } as any);
    const parsed = JSON.parse(result);
    expect(parsed).toHaveProperty("error");
  });

  test("lspTool.execute - 文件路径错误应返回错误", async () => {
    const result = await lspTool.execute({
      action: "gotoDefinition",
      filePath: "/nonexistent/file.ts",
      line: 1,
      character: 0
    } as any);
    const parsed = JSON.parse(result);
    expect(parsed).toHaveProperty("error");
  });
});
```
