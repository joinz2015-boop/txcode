## 3. 集成测试

### 3.1 服务器启动集成测试

**文件**: `test/lsp/startup.test.ts`

```typescript
import { describe, test, expect, beforeAll, afterAll } from "bun:test";
import { LSPManager } from "../../src/lsp/manager";
import { LSPDownloader } from "../../src/lsp/downloader";
import { LSPClient } from "../../src/lsp/client";

describe("LSP 服务器启动集成测试", () => {
  beforeAll(async () => {
    await LSPManager.initDatabase();
  });

  afterAll(async () => {
    // 清理所有 LSP 服务器
    await LSPManager.stopAll();
  });

  test("应下载并启动 Java 服务器", async () => {
    // 确保 Java 服务器已启用
    await LSPManager.updateServer("java", { enabled: true });
    
    // 检查是否需要下载
    const needsDownload = await LSPDownloader.needsDownload("java");
    
    if (needsDownload) {
      // 下载
      const downloaded = await LSPDownloader.download("java");
      expect(downloaded).toBe(true);
    }
    
    // 启动服务器
    const started = await LSPManager.startServer("java");
    
    if (started) {
      // 验证服务器运行状态
      const status = await LSPManager.getServerStatus("java");
      expect(status.state).toBe("running");
    }
  }, 120000);

  test("应能停止服务器", async () => {
    await LSPManager.startServer("typescript");
    await LSPManager.stopServer("typescript");
    
    const status = await LSPManager.getServerStatus("typescript");
    expect(status.state).toBe("stopped");
  });

  test("Java 版本不足 21 应拒绝启动", async () => {
    // 确保 Java 服务器已启用
    await LSPManager.updateServer("java", { enabled: true });
    
    // 检查 Java 版本
    const javaVersion = await LSPManager.checkJavaVersion();
    
    if (!javaVersion.valid || (javaVersion.version && javaVersion.version < 21)) {
      // 版本不足，尝试启动应失败
      const started = await LSPManager.startServer("java");
      expect(started).toBe(false);
      
      // 状态应为版本错误
      const status = await LSPManager.getServerStatus("java");
      expect(status.state).toBe("version_error");
    }
    // 如果版本满足要求，则正常启动
  });
});
```

### 3.2 Web API 集成测试

**文件**: `test/api/lsp.test.ts`

```typescript
import { describe, test, expect, beforeAll, afterAll } from "bun:test";
import { Application } from "../../src/api";

describe("LSP API", () => {
  let app: Application;
  
  beforeAll(async () => {
    app = await Application.start({ port: 3001 });
  });

  afterAll(async () => {
    await app.stop();
  });

  test("GET /api/lsp/servers - 应返回所有服务器", async () => {
    const response = await fetch("http://localhost:3001/api/lsp/servers");
    const servers = await response.json();
    
    expect(Array.isArray(servers)).toBe(true);
    expect(servers.length).toBeGreaterThan(0);
    
    const ts = servers.find((s: any) => s.id === "typescript");
    expect(ts).toHaveProperty("enabled");
    expect(ts).toHaveProperty("status");
  });

  test("PUT /api/lsp/servers/:id - 应更新配置", async () => {
    const response = await fetch("http://localhost:3001/api/lsp/servers/typescript", {
      method: "PUT",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ enabled: false })
    });
    
    expect(response.ok).toBe(true);
  });

  test("POST /api/lsp/servers/:id/start - 应启动服务器", async () => {
    // 先启用
    await fetch("http://localhost:3001/api/lsp/servers/typescript", {
      method: "PUT",
      body: JSON.stringify({ enabled: true })
    });
    
    const response = await fetch("http://localhost:3001/api/lsp/servers/typescript/start", {
      method: "POST"
    });
    
    const result = await response.json();
    expect(result).toHaveProperty("success");
  });

  test("POST /api/lsp/servers/:id/stop - 应停止服务器", async () => {
    const response = await fetch("http://localhost:3001/api/lsp/servers/typescript/stop", {
      method: "POST"
    });
    
    expect(response.ok).toBe(true);
  });

  test("GET /api/lsp/servers/:id/status - 应返回状态", async () => {
    const response = await fetch("http://localhost:3001/api/lsp/servers/typescript/status");
    const status = await response.json();
    
    expect(status).toHaveProperty("state");
    expect(status).toHaveProperty("pid");
    expect(typeof status.running).toBe("boolean");
  });

  test("GET /api/lsp/java-version - 应返回 Java 版本信息", async () => {
    const response = await fetch("http://localhost:3001/api/lsp/java-version");
    const result = await response.json();
    
    expect(result).toHaveProperty("valid");
    expect(result).toHaveProperty("version");
    expect(result).toHaveProperty("error");
  });
});
```

test.describe("LSP 设置页面", () => {
  test.beforeEach(async ({ page }) => {
    await page.goto("/settings/lsp");
  });

    await page.reload();
    
    const cppCheckboxAfter = page.locator("[data-lsp-server='cpp'] input[type='checkbox']");

  test("应显示服务器状态", async ({ page }) => {
    const status = page.locator("[data-lsp-server='typescript'] [data-status]");

## 6. 性能测试

### 6.1 并发启动性能

```typescript
test("5 个服务器并发启动应在 5 秒内完成", async () => {
  const servers = ["typescript", "python", "java", "rust", "go"];
  
  // 先确保所有服务器都已安装
  for (const id of servers) {
    if (await LSPDownloader.needsDownload(id)) {
      await LSPDownloader.download(id);
    }
  }
  
  const start = Date.now();
  
  await Promise.all(
    servers.map(id => LSPManager.startServer(id))
  );
  
  const duration = Date.now() - start;
  
  expect(duration).toBeLessThan(5000);
});
```
